-- BELOW DATABASE CREATED
CREATE TABLE Customer(
    CustomerID CHAR(10) PRIMARY KEY,
    FirstName CHAR(30),
    LastName CHAR(30),
    Contact INT
);

--BELOW DATABASE CREATED
CREATE TABLE PersonalInfo(
    CustomerID CHAR(10) PRIMARY KEY,
    Email CHAR(50),
    Phone INT,
    Address CHAR(255),
    City CHAR(20),
    PostalCode INT,
    Country CHAR(20)
);

-- BELOW DATABASE CREATED
CREATE TABLE BillingInfo(
    BillingID CHAR(10) PRIMARY KEY,
    BillDate TIMESTAMP,
    BillingAddress CHAR(255),
    CreditCardID INT,
    CreditCardPIN INT,
    CreditCardExpDate CHAR(7),
    CreditCardNo INT
);

-- BELOW DATABASE CREATED
CREATE TABLE Order_(
    OrderID CHAR(10) PRIMARY KEY,
    CustomerID CHAR(10),
    ProductID CHAR(10),
    OrderDate TIMESTAMP,
    UnitPrice FLOAT(6,2),
    RequiredDate TIMESTAMP,
    Quantity INT,
    OrderNumber CHAR(10),
    Discount FLOAT(3,2),
    Amount FLOAT(6,2)
);

-- BELOW DATABASE CREATED
CREATE TABLE Cart(
    CartID CHAR(10) PRIMARY KEY,
    NumberOfProducts INT,
    TotalPrice FLOAT(6,2)
);

-- BELOW DATABASE CREATED BUT WITHOUT IMAGE
CREATE TABLE Product(
    ProductID CHAR(10) PRIMARY KEY,
    ProductName CHAR(100),
    ProductDescription CHAR(255),
    UnitPrice FLOAT(6,2),
    Category_ CHAR(20),
    Discount FLOAT(3,2),
    UnitsOnOrder INT,
    UnitsInStock INT
);

-- BELOW DATABASE CREATED
CREATE TABLE Admin_(
    AdminID CHAR(10) PRIMARY KEY,
    AdminFname CHAR(30),
    AdminLname CHAR(30),
    AdminPassword CHAR(50)
);
